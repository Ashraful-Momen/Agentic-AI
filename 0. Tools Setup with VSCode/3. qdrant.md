#install the image with the docker and run the container 

qdrant is the vector DB that need to run with local and connect with the roo code in the index . 



--------------------------------------------------

mkdir -p ./qdrant_storage
docker run -d --name qdrant \
  -p 6333:6333 -p 6334:6334 \
  -v "$(pwd)/qdrant_storage":/qdrant/storage \
  -e QDRANT__SERVICE__HTTP_PORT=6333 \
  -e QDRANT__SERVICE__GRPC_PORT=6334 \
  qdrant/qdrant:latest


----------------------------------------------------

after run the qdrant then set the qdrant into the index(prompt box bd incon) option . 

#docker-compose.yml => alwasy run the qdrant 
-------------------
version: '3.8'

services:
  qdrant:
    image: qdrant/qdrant:latest
    container_name: qdrant-db
    restart: always
    ports:
      - "6333:6333"
      - "6334:6334"
    volumes:
      - ~/Software/qdrant_storage:/qdrant/storage
    environment:
      - QDRANT__SERVICE__HTTP_PORT=6333
      - QDRANT__SERVICE__GRPC_PORT=6334
    networks:
      - qdrant-network

networks:
  qdrant-network:
    driver: bridge
